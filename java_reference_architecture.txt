// --- Service: CurrencyService.java ---
/*
package com.converter.service;

import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import com.google.gson.JsonObject;
import com.google.gson.JsonParser;

public class CurrencyService {
    private static final String API_KEY = "YOUR_API_KEY_HERE";
    private static final String BASE_URL = "https://v6.exchangerate-api.com/v6/";

    public double getExchangeRate(String from, String to) throws Exception {
        HttpClient client = HttpClient.newHttpClient();
        HttpRequest request = HttpRequest.newBuilder()
            .uri(java.net.URI.create(BASE_URL + API_KEY + "/pair/" + from + "/" + to))
            .build();

        HttpResponse<String> response = client.send(request, HttpResponse.BodyHandlers.ofString());
        JsonObject json = JsonParser.parseString(response.body()).getAsJsonObject();
        
        return json.get("conversion_rate").getAsDouble();
    }
}
*/

// --- Controller: ConverterController.java ---
/*
package com.converter.controller;

import javafx.fxml.FXML;
import javafx.scene.control.*;
import com.converter.service.CurrencyService;

public class ConverterController {
    @FXML private TextField amountInput;
    @FXML private ComboBox<String> fromCombo;
    @FXML private ComboBox<String> toCombo;
    @FXML private Label resultLabel;

    private final CurrencyService service = new CurrencyService();

    @FXML
    public void handleConvert() {
        try {
            double amount = Double.parseDouble(amountInput.getText());
            double rate = service.getExchangeRate(fromCombo.getValue(), toCombo.getValue());
            resultLabel.setText(String.format("%.2f", amount * rate));
        } catch (Exception e) {
            System.err.println("Conversion Error: " + e.getMessage());
        }
    }
    
    @FXML
    public void handleReverse() {
        String from = fromCombo.getValue();
        fromCombo.setValue(toCombo.getValue());
        toCombo.setValue(from);
    }
}
*/
